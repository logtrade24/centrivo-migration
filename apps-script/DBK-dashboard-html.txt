/**
 * Renderuje historię nieobecności
 */
function renderAbsenceHistory(history, selectedDays = null) {
  console.log('=== RENDEROWANIE HISTORII ===');
  console.log('Liczba nieobecności:', history.absences.length);
  console.log('Pierwsze 5 nieobecności:', history.absences.slice(0, 5));
  console.log('============================');
  console.log('Historia otrzymana:', history);
  
  let html = '';
  
  // Filtry
  html += `
  <div class="absence-filters">
    <select id="absencePeriodFilter">
  <option value="7" ${selectedDays === 7 ? 'selected' : ''}>Ostatnie 7 dni</option>
  <option value="14" ${selectedDays === 14 ? 'selected' : ''}>Ostatnie 14 dni</option>
  <option value="30" ${selectedDays === 30 ? 'selected' : ''}>Ostatnie 30 dni</option>
  <option value="60" ${selectedDays === 60 ? 'selected' : ''}>Ostatnie 60 dni</option>
  <option value="90" ${selectedDays === 90 ? 'selected' : ''}>Ostatnie 90 dni</option>
  <option value="null" ${selectedDays === null ? 'selected' : ''}>Wszystkie (z arkusza Logi)</option>
</select>
    
    <button onclick="refreshAbsenceHistory()" class="export-btn">
      🔄 Odśwież historię
    </button>
      
      <button onclick="exportAbsenceToExcel()" class="export-btn">
        📥 Eksport do Excel
      </button>
      
      <button onclick="exportAbsenceToPDF()" class="export-btn">
        📄 Eksport do PDF
      </button>
    </div>
  `;
  
  // Tabela
  html += '<table class="absence-history-table">';
  html += `
    <thead>
      <tr>
        <th>Data</th>
        <th>Dni</th>
        <th>Pracownik</th>
        <th>Typ</th>
        <th>Szczegóły</th>
        <th>Status</th>
      </tr>
    </thead>
    <tbody>
  `;
  
  // Dzisiejsza data
const today = new Date();
today.setHours(0, 0, 0, 0);

history.absences.forEach(absence => {
  const fromDate = new Date(absence.dateFrom);
  fromDate.setHours(0, 0, 0, 0);
  const toDate = new Date(absence.dateTo);
  toDate.setHours(0, 0, 0, 0);
  
  // Pogrubienie tylko jeśli trwa DZIŚ
  const isOngoingToday = today >= fromDate && today <= toDate;
    
    const rowClass = `absence-${absence.type}${isOngoingToday ? ' current-week' : ''}`;
    
    const dateRange = absence.dateFrom === absence.dateTo 
      ? formatDate(absence.dateFrom)
      : `${formatDate(absence.dateFrom)}-${formatDate(absence.dateTo)}`;
    
    const daysText = absence.days > 0 ? absence.days : '-';
    
    html += `
      <tr class="${rowClass}">
        <td>${dateRange}</td>
        <td>${daysText}</td>
        <td>${absence.name}</td>
        <td>${absence.typeText}</td>
        <td>${absence.details}</td>
        <td>${absence.status}</td>
      </tr>
    `;
  });
  
  html += '</tbody></table>';
  
  // Statystyki
  const startDate = new Date(history.stats.periodStart);
  const endDate = new Date(history.stats.periodEnd);
  const daysDiff = Math.ceil((endDate - startDate) / (1000 * 60 * 60 * 24));

  html += `
    <div class="absence-stats">
      <h3>📊 PODSUMOWANIE (ostatnie ${daysDiff} dni):</h3>
      <p>
        • Dni urlopowe: <strong>${history.stats.totalVacationDays}</strong> &nbsp;&nbsp;
        • Dni L4: <strong>${history.stats.totalL4Days}</strong> &nbsp;&nbsp;
        • Dni wolne: <strong>${history.stats.totalDayOffDays || 0}</strong> &nbsp;&nbsp;
        • Urlopy okolicznościowe: <strong>${history.stats.totalOccasionalDays || 0}</strong> &nbsp;&nbsp;
        • Dni nieusprawiedliwione: <strong>${history.stats.totalUnjustifiedDays}</strong>
      </p>
    </div>
  `;
  
  document.getElementById('absence-history-container').innerHTML = html;
}

/**
 * Pomocnicze funkcje
 */
function getMonday(date) {
  const d = new Date(date);
  const day = d.getDay();
  const diff = d.getDate() - day + (day === 0 ? -6 : 1);
  return new Date(d.setDate(diff));
}

function formatDate(dateStr) {
  const parts = dateStr.split('-');
  return `${parts[2]}.${parts[1]}.${parts[0]}`;
}

function exportAbsenceToExcel() {
  const select = document.getElementById('absencePeriodFilter');
  const days = select.value === 'null' ? null : parseInt(select.value);
  
  const btn = event.target;
  const originalText = btn.textContent;
  btn.textContent = '⏳ Generowanie Excel...';
  btn.disabled = true;
  
  console.log('Rozpoczynam eksport do Excel, dni:', days);
  
  google.script.run
    .withSuccessHandler(function(result) {
      console.log('Wynik eksportu Excel:', result);
      
      btn.textContent = originalText;
      btn.disabled = false;
      
      if (result.success) {
        // Otwórz plik do pobrania
        const downloadWindow = window.open(result.downloadUrl, '_blank');
        
        if (!downloadWindow) {
          // Popup zablokowany - pokaż link
          showDownloadLink(result, 'Excel');
        } else {
          showNotification(`✅ Excel wygenerowany: ${result.fileName}`, 'success');
        }
      } else {
        showNotification(`❌ ${result.message}`, 'error');
      }
    })
    .withFailureHandler(function(error) {
      console.error('Błąd eksportu Excel:', error);
      
      btn.textContent = originalText;
      btn.disabled = false;
      
      showNotification('❌ Błąd: ' + error.message, 'error');
    })
    .exportAbsenceHistoryToExcel(days);
}
