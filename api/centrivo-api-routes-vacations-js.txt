const express = require("express");
const router = express.Router();

const {
  getUserVacationBalance,
  createVacationRequest,
  listMyVacationRequests,
  listPendingVacationRequests,
  processVacationRequest,
  submitYearPlanRequest,
  listMyYearPlanRequests,
  listPendingYearPlanRequests,
  processYearPlanRequest,
  processYearPlanWithDates,
  getYearPlanDatesForCalendar,
  getPendingAbsenceCountsByType,
  getPendingAbsenceRequestsCount,
  getUserAllAbsenceHistory,
} = require("../services/sheets");

function isManager(loginRaw) {
  const login = String(loginRaw || "").trim().toLowerCase();
  const managers = String(process.env.MANAGER_LOGINS || "jerzyd")
    .split(",")
    .map(s => s.trim().toLowerCase())
    .filter(Boolean);
  return managers.includes(login);
}

// ===== URLOPY WYPOCZYNKOWE =====

// POST /vacations/balance { login, code? }
router.post("/balance", async (req, res) => {
  try {
    const login = String(req.body?.login || "").trim();
    const code  = String(req.body?.code  || "").trim();
    if (!login) return res.status(400).json({ success: false, message: "Brak login" });

    const balance = await getUserVacationBalance(login, code);
    return res.json({ success: true, balance });
  } catch (e) {
    console.error("VACATIONS /balance ERROR:", e);
    return res.status(500).json({ success: false, message: "Błąd serwera" });
  }
});

// POST /vacations/create { login, startDate, endDate, comment?, vacationType?, code? }
router.post("/create", async (req, res) => {
  try {
    const login = String(req.body?.login || "").trim();
    const code  = String(req.body?.code  || "").trim();
    const startDate = String(req.body?.startDate || "").trim();
    const endDate = String(req.body?.endDate || "").trim();
    const comment = String(req.body?.comment || "").trim();
    const vacationType = String(req.body?.vacationType || "planowany").trim();

    if (!login) return res.status(400).json({ success: false, message: "Brak login" });
    if (!startDate || !endDate) return res.status(400).json({ success: false, message: "Brak dat" });

    const result = await createVacationRequest(login, startDate, endDate, comment, vacationType, code);
    if (!result?.success) return res.status(400).json(result);
    return res.json(result);
  } catch (e) {
    console.error("VACATIONS /create ERROR:", e);
    return res.status(500).json({ success: false, message: "Błąd serwera" });
  }
});

// POST /vacations/pending { login }  (kierownik)
// Zwraca oczekujące wnioski + informację o kolizjach z zatwierdzonymi urlopami

  }
});

// POST /vacations/process { login, requestId, decision, comment? }  (kierownik)
// decision: "Zatwierdzony" | "Odrzucony"
// Przy zatwierdzeniu automatycznie aktualizuje saldo urlopowe pracownika
router.post("/process", async (req, res) => {
  try {
    const login = String(req.body?.login || "").trim().toLowerCase();
    const requestId = Number(req.body?.requestId || 0);
    const decision = String(req.body?.decision || "").trim();
    const comment = String(req.body?.comment || "").trim();

    if (!login) return res.status(400).json({ success: false, message: "Brak login" });
    if (!isManager(login)) return res.status(403).json({ success: false, message: "Brak uprawnień" });
    if (!requestId) return res.status(400).json({ success: false, message: "Brak requestId" });
    if (!["Zatwierdzony", "Odrzucony"].includes(decision)) {
      return res.status(400).json({ success: false, message: "Zła decyzja (Zatwierdzony / Odrzucony)" });
    }

    const result = await processVacationRequest(requestId, decision, comment);
    if (!result?.success) return res.status(400).json(result);
    return res.json(result);
  } catch (e) {
    console.error("VACATIONS /process ERROR:", e);
    return res.status(500).json({ success: false, message: "Błąd serwera" });
  }
});

// POST /vacations/year-plan/submit { login, year, dates?, newDates?, removedDates?, name? }
// Pracownik składa lub aktualizuje swój plan roczny
router.post("/year-plan/submit", async (req, res) => {
  try {
    const login = String(req.body?.login || "").trim();
    const year = Number(req.body?.year || new Date().getFullYear());
    const name = String(req.body?.name || "").trim();
    const dates = req.body?.dates;
    const newDates = req.body?.newDates;
    const removedDates = req.body?.removedDates;

    if (!login) return res.status(400).json({ success: false, message: "Brak login" });
    if (!year || year < 2024 || year > 2030) {
      return res.status(400).json({ success: false, message: "Nieprawidłowy rok" });
    }

    const result = await submitYearPlanRequest({ login, year, name, dates, newDates, removedDates });
    if (!result?.success) return res.status(400).json(result);
    return res.json(result);
  } catch (e) {
    console.error("VACATIONS /year-plan/submit ERROR:", e);
    return res.status(500).json({ success: false, message: "Błąd serwera" });
  }
});

// POST /vacations/year-plan/process-dates { login, requestId, decision, comment?, selectedDates }  (kierownik)
// Częściowe zatwierdzenie — kierownik wskazuje które dni zatwierdza
// decision: "Zatwierdzony" | "Częściowo zatwierdzony" | "Odrzucony"
router.post("/year-plan/process-dates", async (req, res) => {
  try {
    const login = String(req.body?.login || "").trim().toLowerCase();
    const requestId = Number(req.body?.requestId || 0);
    const decision = String(req.body?.decision || "").trim();
    const comment = String(req.body?.comment || "").trim();
    const selectedDates = Array.isArray(req.body?.selectedDates) ? req.body.selectedDates : [];

    if (!login) return res.status(400).json({ success: false, message: "Brak login" });
    if (!isManager(login)) return res.status(403).json({ success: false, message: "Brak uprawnień" });
    if (!requestId) return res.status(400).json({ success: false, message: "Brak requestId" });
    if (!["Zatwierdzony", "Częściowo zatwierdzony", "Odrzucony"].includes(decision)) {
      return res.status(400).json({ success: false, message: "Zła decyzja" });
    }
    if (decision !== "Odrzucony" && selectedDates.length === 0) {
      return res.status(400).json({ success: false, message: "Brak selectedDates dla zatwierdzenia" });
    }

    const result = await processYearPlanWithDates(requestId, decision, comment, selectedDates);
    if (!result?.success) return res.status(400).json(result);
    return res.json(result);
  } catch (e) {
    console.error("VACATIONS /year-plan/process-dates ERROR:", e);
    return res.status(500).json({ success: false, message: "Błąd serwera" });
  }
});
